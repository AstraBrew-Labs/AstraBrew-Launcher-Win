### `parse_tavern_config_payload`（第 1324 行）

将 `config.yaml` 的原始字符串解析为 `TavernConfigPayload` 结构体。

```rust
fn parse_tavern_config_payload(yaml_str: &str) -> Result<TavernConfigPayload, String> {
    let root: serde_yaml::Value =
        serde_yaml::from_str(yaml_str).map_err(|e| format!("解析配置失败: {}", e))?;
    let mapping = root
        .as_mapping()
        .ok_or("配置文件格式无效，根节点必须是对象".to_string())?;

    let get_bool = |key: &str, default: bool| {
        mapping
            .get(serde_yaml::Value::String(key.to_string()))
            .and_then(serde_yaml::Value::as_bool)
            .unwrap_or(default)
    };
    let get_i64 = |key: &str, default: i64| {
        mapping
            .get(serde_yaml::Value::String(key.to_string()))
            .and_then(serde_yaml::Value::as_i64)
            .unwrap_or(default)
    };
    let parse_str_seq = |value: Option<&serde_yaml::Value>, default: Vec<String>| -> Vec<String> {
        value
            .and_then(serde_yaml::Value::as_sequence)
            .map(|seq| {
                seq.iter()
                    .filter_map(serde_yaml::Value::as_str)
                    .map(|s| s.to_string())
                    .collect()
            })
            .unwrap_or(default)
    };
    let parse_i64_seq = |value: Option<&serde_yaml::Value>, default: Vec<i64>| -> Vec<i64> {
        value
            .and_then(serde_yaml::Value::as_sequence)
            .map(|seq| {
                seq.iter()
                    .filter_map(serde_yaml::Value::as_i64)
                    .collect::<Vec<_>>()
            })
            .filter(|s| !s.is_empty())
            .unwrap_or(default)
    };
    let key = |s: &str| serde_yaml::Value::String(s.to_string());
    let sub = |k: &str| {
        mapping
            .get(serde_yaml::Value::String(k.to_string()))
            .and_then(serde_yaml::Value::as_mapping)
    };

    // ── listenAddress ──
    let listen_address = sub("listenAddress")
        .map(|m| TavernDualStackAddress {
            ipv4: m
                .get(key("ipv4"))
                .and_then(serde_yaml::Value::as_str)
                .unwrap_or("0.0.0.0")
                .to_string(),
            ipv6: m
                .get(key("ipv6"))
                .and_then(serde_yaml::Value::as_str)
                .unwrap_or("[::]")
                .to_string(),
        })
        .unwrap_or(TavernDualStackAddress {
            ipv4: "0.0.0.0".to_string(),
            ipv6: "[::]".to_string(),
        });

    // ── protocol ──
    let protocol = sub("protocol")
        .map(|m| TavernDualStackProtocol {
            ipv4: m
                .get(key("ipv4"))
                .and_then(serde_yaml::Value::as_bool)
                .unwrap_or(true),
            ipv6: m
                .get(key("ipv6"))
                .and_then(serde_yaml::Value::as_bool)
                .unwrap_or(false),
        })
        .unwrap_or(TavernDualStackProtocol {
            ipv4: true,
            ipv6: false,
        });

    // ── whitelist ──
    let whitelist = parse_str_seq(
        mapping.get(key("whitelist")),
        vec!["::1".to_string(), "127.0.0.1".to_string()],
    );

    // ── basicAuthUser ──
    let basic_auth_user = sub("basicAuthUser")
        .map(|m| TavernBasicAuthUser {
            username: m
                .get(key("username"))
                .and_then(serde_yaml::Value::as_str)
                .unwrap_or("user")
                .to_string(),
            password: m
                .get(key("password"))
                .and_then(serde_yaml::Value::as_str)
                .unwrap_or("password")
                .to_string(),
        })
        .unwrap_or(TavernBasicAuthUser {
            username: "user".to_string(),
            password: "password".to_string(),
        });

    // ── cors ──
    let cors = sub("cors")
        .map(|m| TavernCorsConfig {
            enabled: m
                .get(key("enabled"))
                .and_then(serde_yaml::Value::as_bool)
                .unwrap_or(true),
            origin: parse_str_seq(m.get(key("origin")), vec!["null".to_string()]),
            methods: parse_str_seq(m.get(key("methods")), vec!["OPTIONS".to_string()]),
            allowed_headers: parse_str_seq(m.get(key("allowedHeaders")), vec![]),
            exposed_headers: parse_str_seq(m.get(key("exposedHeaders")), vec![]),
            credentials: m
                .get(key("credentials"))
                .and_then(serde_yaml::Value::as_bool)
                .unwrap_or(false),
            max_age: m.get(key("maxAge")).and_then(serde_yaml::Value::as_i64),
        })
        .unwrap_or(TavernCorsConfig {
            enabled: true,
            origin: vec!["null".to_string()],
            methods: vec!["OPTIONS".to_string()],
            allowed_headers: vec![],
            exposed_headers: vec![],
            credentials: false,
            max_age: None,
        });

    // ── requestProxy ──
    let request_proxy = sub("requestProxy")
        .map(|m| TavernRequestProxyConfig {
            enabled: m
                .get(key("enabled"))
                .and_then(serde_yaml::Value::as_bool)
                .unwrap_or(false),
            url: m
                .get(key("url"))
                .and_then(serde_yaml::Value::as_str)
                .unwrap_or("")
                .to_string(),
            bypass: m
                .get(key("bypass"))
                .and_then(serde_yaml::Value::as_sequence)
                .map(|s| {
                    s.iter()
                        .filter_map(serde_yaml::Value::as_str)
                        .map(|s| s.to_string())
                        .collect()
                })
                .unwrap_or_default(),
        })
        .unwrap_or(TavernRequestProxyConfig {
            enabled: false,
            url: "".to_string(),
            bypass: vec![],
        });

    // ── backups ──
    let backups = sub("backups")
        .map(|bk| {
            let common = child_map(bk, "common")
                .ok()
                .and_then(|bc| bc.get(key("numberOfBackups")))
                .and_then(serde_yaml::Value::as_i64)
                .unwrap_or(5);
            let chat = child_map(bk, "chat").ok();
            TavernBackupsConfig {
                common: TavernBackupCommonConfig {
                    number_of_backups: common,
                },
                chat: TavernBackupChatConfig {
                    enabled: chat
                        .and_then(|c| c.get(key("enabled")))
                        .and_then(serde_yaml::Value::as_bool)
                        .unwrap_or(true),
                    check_integrity: chat
                        .and_then(|c| c.get(key("checkIntegrity")))
                        .and_then(serde_yaml::Value::as_bool)
                        .unwrap_or(true),
                    max_total_backups: chat
                        .and_then(|c| c.get(key("maxTotalBackups")))
                        .and_then(serde_yaml::Value::as_i64)
                        .unwrap_or(200),
                    throttle_interval: chat
                        .and_then(|c| c.get(key("throttleInterval")))
                        .and_then(serde_yaml::Value::as_i64)
                        .unwrap_or(30000),
                },
            }
        })
        .unwrap_or_default();

    // ── thumbnails ──
    let thumbnails = sub("thumbnails")
        .map(|th| {
            let dim = child_map(th, "dimensions").ok();
            TavernThumbnailsConfig {
                enabled: th
                    .get(key("enabled"))
                    .and_then(serde_yaml::Value::as_bool)
                    .unwrap_or(true),
                format: th
                    .get(key("format"))
                    .and_then(serde_yaml::Value::as_str)
                    .unwrap_or("webp")
                    .to_string(),
                quality: th
                    .get(key("quality"))
                    .and_then(serde_yaml::Value::as_i64)
                    .unwrap_or(75),
                dimensions: TavernThumbnailDimensions {
                    bg: dim
                        .and_then(|d| d.get(key("bg")))
                        .and_then(serde_yaml::Value::as_sequence)
                        .map(|s| s.iter().filter_map(serde_yaml::Value::as_i64).collect())
                        .unwrap_or(vec![400, 600]),
                    avatar: dim
                        .and_then(|d| d.get(key("avatar")))
                        .and_then(serde_yaml::Value::as_sequence)
                        .map(|s| s.iter().filter_map(serde_yaml::Value::as_i64).collect())
                        .unwrap_or(vec![100, 100]),
                    persona: dim
                        .and_then(|d| d.get(key("persona")))
                        .and_then(serde_yaml::Value::as_sequence)
                        .map(|s| s.iter().filter_map(serde_yaml::Value::as_i64).collect())
                        .unwrap_or(vec![400, 600]),
                },
            }
        })
        .unwrap_or_default();

    // ── browserLaunch ──
    let browser_launch = sub("browserLaunch")
        .map(|bl| TavernBrowserLaunchConfig {
            enabled: bl
                .get(key("enabled"))
                .and_then(serde_yaml::Value::as_bool)
                .unwrap_or(false),
            browser: bl
                .get(key("browser"))
                .and_then(serde_yaml::Value::as_str)
                .unwrap_or("chrome")
                .to_string(),
        })
        .unwrap_or_default();

    // ── ssl ── (仅全局配置)
    let ssl = sub("ssl")
        .map(|s| TavernSslConfig {
            enabled: s
                .get(key("enabled"))
                .and_then(serde_yaml::Value::as_bool)
                .unwrap_or(false),
            cert_path: s
                .get(key("certPath"))
                .and_then(serde_yaml::Value::as_str)
                .unwrap_or("")
                .to_string(),
            key_path: s
                .get(key("keyPath"))
                .and_then(serde_yaml::Value::as_str)
                .unwrap_or("")
                .to_string(),
            key_passphrase: s
                .get(key("keyPassphrase"))
                .and_then(serde_yaml::Value::as_str)
                .unwrap_or("")
                .to_string(),
        })
        .unwrap_or_default();

    Ok(TavernConfigPayload {
        port: get_i64("port", 8000) as u16,
        listen: get_bool("listen", true),
        listen_address,
        protocol,
        basic_auth_mode: get_bool("basicAuthMode", false),
        enable_user_accounts: get_bool("enableUserAccounts", false),
        enable_discreet_login: get_bool("enableDiscreetLogin", false),
        per_user_basic_auth: get_bool("perUserBasicAuth", false),
        basic_auth_user,
        whitelist_mode: get_bool("whitelistMode", false),
        whitelist,
        cors,
        request_proxy,
        backups,
        thumbnails,
        browser_launch_enabled: browser_launch.enabled,
        browser_type: browser_launch.browser,
        // 以下为全局配置额外字段
        ssl,
        dns_prefer_ipv6: get_bool("dnsPreferIPv6", false),
        heartbeat_interval: get_i64("heartbeatInterval", 0) as u32,
        host_whitelist: parse_host_whitelist(mapping),
        whitelist_import_domains: parse_str_seq(
            mapping.get(key("whitelistImportDomains")),
            vec![],
        ),
        // ... session, security, etc.
    })
}
```

## 五、更新配置选项

### `update_sillytavern_config_options`（第 1812 行）

接收 `TavernConfigPayload`，写回 `config.yaml`。

核心逻辑：
1. 读取现有 YAML → `serde_yaml::Value`
2. 用 `upsert()` 逐字段更新
3. 用 `child_map()` 获取或创建子 Mapping
4. `serde_yaml::to_string()` 序列化写回

```rust
#[tauri::command]
pub async fn update_sillytavern_config_options(
    app: AppHandle,
    version: String,
    config: TavernConfigPayload,
) -> Result<TavernConfigPayload, String> {
    let lang = get_current_lang(&app);
    let app2 = app.clone();
    tokio::task::spawn_blocking(move || {
        let path = get_st_config_path(&app2, &version)?;
        let content = fs::read_to_string(&path).map_err(|e| match lang {
            Lang::ZhCn => format!("读取失败: {}", e),
            Lang::EnUs => format!("Read failed: {}", e),
        })?;
        let mut root: serde_yaml::Value =
            serde_yaml::from_str(&content).map_err(|e| match lang {
                Lang::ZhCn => format!("解析配置失败: {}", e),
                Lang::EnUs => format!("Parse failed: {}", e),
            })?;
        let m = root
            .as_mapping_mut()
            .ok_or("配置文件格式无效，根节点必须是对象".to_string())?;

        // 顶层字段
        upsert(m, "port", serde_yaml::Value::Number(serde_yaml::Number::from(config.port)));
        upsert(m, "listen", serde_yaml::Value::Bool(config.listen));

        // listenAddress（子 Mapping）
        {
            let la = child_map(m, "listenAddress")?;
            upsert(la, "ipv4", serde_yaml::Value::String(config.listen_address.ipv4.clone()));
            upsert(la, "ipv6", serde_yaml::Value::String(config.listen_address.ipv6.clone()));
        }
        // protocol（子 Mapping）
        {
            let p = child_map(m, "protocol")?;
            upsert(p, "ipv4", serde_yaml::Value::Bool(config.protocol.ipv4));
            upsert(p, "ipv6", serde_yaml::Value::Bool(config.protocol.ipv6));
        }

        upsert(m, "basicAuthMode", serde_yaml::Value::Bool(config.basic_auth_mode));
        upsert(m, "enableUserAccounts", serde_yaml::Value::Bool(config.enable_user_accounts));
        upsert(m, "enableDiscreetLogin", serde_yaml::Value::Bool(config.enable_discreet_login));
        upsert(m, "perUserBasicAuth", serde_yaml::Value::Bool(config.per_user_basic_auth));

        // basicAuthUser（子 Mapping）
        {
            let bau = child_map(m, "basicAuthUser")?;
            upsert(bau, "username", serde_yaml::Value::String(config.basic_auth_user.username.clone()));
            upsert(bau, "password", serde_yaml::Value::String(config.basic_auth_user.password.clone()));
        }

        upsert(m, "whitelistMode", serde_yaml::Value::Bool(config.whitelist_mode));
        upsert(m, "whitelist",
            serde_yaml::Value::Sequence(
                config.whitelist.iter().map(|s| serde_yaml::Value::String(s.clone())).collect()
            )
        );

        // cors（子 Mapping）
        {
            let c = child_map(m, "cors")?;
            upsert(c, "enabled", serde_yaml::Value::Bool(config.cors.enabled));
            upsert(c, "origin", /* Sequence */);
            upsert(c, "methods", /* Sequence */);
            upsert(c, "allowedHeaders", /* Sequence */);
            upsert(c, "exposedHeaders", /* Sequence */);
            upsert(c, "credentials", serde_yaml::Value::Bool(config.cors.credentials));
            upsert(c, "maxAge", /* Number or Null */);
        }

        // requestProxy（子 Mapping）
        {
            let rp = child_map(m, "requestProxy")?;
            upsert(rp, "enabled", serde_yaml::Value::Bool(config.request_proxy.enabled));
            upsert(rp, "url", serde_yaml::Value::String(config.request_proxy.url.clone()));
            upsert(rp, "bypass", /* Sequence */);
        }

        // backups / thumbnails / browserLaunch ...
        // （完整代码见源文件第 1998~2120 行）

        let new_content = serde_yaml::to_string(&root).map_err(|e| match lang {
            Lang::ZhCn => format!("序列化配置失败: {}", e),
            Lang::EnUs => format!("Serialize failed: {}", e),
        })?;
        fs::write(&path, new_content).map_err(|e| match lang {
            Lang::ZhCn => format!("写入失败: {}", e),
            Lang::EnUs => format!("Write failed: {}", e),
        })?;
        Ok(config)
    })
    .await
    .map_err(|e| e.to_string())?
}
```

## 六、辅助函数

### `upsert`（更新或插入 YAML 字段）

```rust
fn upsert(mapping: &mut serde_yaml::Mapping, key: &str, value: serde_yaml::Value) {
    mapping.insert(
        serde_yaml::Value::String(key.to_string()),
        value,
    );
}
```

### `child_map`（获取或创建子 Mapping）

```rust
fn child_map<'a>(
    mapping: &'a mut serde_yaml::Mapping,
    key: &str,
) -> Result<&'a mut serde_yaml::Mapping, String> {
    let ykey = serde_yaml::Value::String(key.to_string());
    if !mapping.contains_key(&ykey) {
        mapping.insert(ykey.clone(), serde_yaml::Value::Mapping(serde_yaml::Mapping::new()));
    }
    mapping
        .get_mut(&ykey)
        .and_then(serde_yaml::Value::as_mapping_mut)
        .ok_or_else(|| format!("字段 {} 不是有效的 Mapping", key))
}
```

## 八、调用流程

```
前端（Vue）
  │
  ├── 读取配置
```
